WITH prod_counts AS (
    SELECT order_items.product_id, COUNT(DISTINCT order_items.order_id) AS orders_count
    FROM order_items
    GROUP BY order_items.product_id
)
SELECT pA.product_id AS product_a_id,
       pA.product_name AS product_a_name,
       pB.product_id AS product_b_id,
       pB.product_name AS product_b_name,
       COUNT(DISTINCT oi1.order_id) AS together_count
FROM order_items oi1
JOIN order_items oi2
  ON oi1.order_id = oi2.order_id
  AND oi1.product_id < oi2.product_id
JOIN prod_counts pcA ON oi1.product_id = pcA.product_id
JOIN prod_counts pcB ON oi2.product_id = pcB.product_id
JOIN products pA ON pcA.product_id = pA.product_id
JOIN products pB ON pcB.product_id = pB.product_id
GROUP BY pA.product_id, pA.product_name, pB.product_id, pB.product_name, pcA.orders_count, pcB.orders_count
HAVING COUNT(DISTINCT oi1.order_id) = pcA.orders_count
   AND COUNT(DISTINCT oi1.order_id) = pcB.orders_count;
