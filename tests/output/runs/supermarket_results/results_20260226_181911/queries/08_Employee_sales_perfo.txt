SELECT 
    e.employee_name,
    COUNT(o.order_id) AS orders_processed,
    COALESCE(SUM(oi.quantity * oi.unit_price), 0) AS total_revenue,
    c.customer_name AS top_customer
FROM employees e
LEFT JOIN orders o ON e.employee_id = o.employee_id
LEFT JOIN order_items oi ON o.order_id = oi.order_id
LEFT JOIN (
    SELECT 
        o.employee_id,
        c.customer_id,
        c.customer_name,
        SUM(oi.quantity * oi.unit_price) AS customer_revenue
    FROM orders o
    JOIN customers c ON o.customer_id = c.customer_id
    JOIN order_items oi ON o.order_id = oi.order_id
    GROUP BY o.employee_id, c.customer_id, c.customer_name
    ORDER BY customer_revenue DESC
) ct ON e.employee_id = ct.employee_id
GROUP BY e.employee_id, e.employee_name, c.customer_name
ORDER BY e.employee_name;
