SELECT 
    c.customer_name,
    AVG(p.spending) AS average_monthly_spending,
    CASE 
        WHEN MAX(p.spending) <= 1.2 * AVG(p.spending) 
             AND MIN(p.spending) >= 0.8 * AVG(p.spending)
        THEN 'Consistent'
        ELSE 'Inconsistent'
    END AS consistency_flag
FROM customers c
JOIN (
    SELECT 
        o.customer_id, 
        DATE_FORMAT(o.order_date, '%Y-%m') AS month, 
        SUM(oi.quantity * oi.unit_price) AS spending
    FROM orders o
    JOIN order_items oi ON o.order_id = oi.order_id
    WHERE YEAR(o.order_date) = 2024
    GROUP BY o.customer_id, DATE_FORMAT(o.order_date, '%Y-%m')
) p ON c.customer_id = p.customer_id
GROUP BY c.customer_id, c.customer_name;
