    c.customer_name AS most_active_customer

SELECT
    DATE_FORMAT(o.order_date, '%Y-%m') AS month,
    SUM(oi.quantity * oi.unit_price) AS total_revenue,
    (
        SELECT p2.product_name
        FROM order_items oi2
        JOIN orders o2 ON oi2.order_id = o2.order_id
        JOIN products p2 ON oi2.product_id = p2.product_id
        WHERE DATE_FORMAT(o2.order_date, '%Y-%m') = DATE_FORMAT(o.order_date, '%Y-%m')
        GROUP BY p2.product_id, p2.product_name
        ORDER BY SUM(oi2.quantity * oi2.unit_price) DESC
        LIMIT 1
    ) AS top_selling_product,
    (
        SELECT e2.employee_name
        FROM orders o3
        JOIN employees e2 ON o3.employee_id = e2.employee_id
        JOIN order_items oi3 ON o3.order_id = oi3.order_id
        WHERE DATE_FORMAT(o3.order_date, '%Y-%m') = DATE_FORMAT(o.order_date, '%Y-%m')
        GROUP BY e2.employee_id, e2.employee_name
        ORDER BY SUM(oi3.quantity * oi3.unit_price) DESC
        LIMIT 1
    ) AS best_performing_employee,
    (
        SELECT c2.customer_name
        FROM orders o4
        JOIN customers c2 ON o4.customer_id = c2.customer_id
        WHERE DATE_FORMAT(o4.order_date, '%Y-%m') = DATE_FORMAT(o.order_date, '%Y-%m')
        GROUP BY c2.customer_id, c2.customer_name
        ORDER BY COUNT(o4.order_id) DESC
        LIMIT 1
    ) AS most_active_customer
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
GROUP BY DATE_FORMAT(o.order_date, '%Y-%m')
ORDER BY month;
