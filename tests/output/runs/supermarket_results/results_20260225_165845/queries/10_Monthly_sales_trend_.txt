WITH month_revenue AS (
  SELECT
    DATE_TRUNC('month', o.order_date) AS month,
    COALESCE(SUM(oi.quantity * oi.unit_price), 0) AS total_revenue
  FROM orders o
  LEFT JOIN order_items oi ON o.order_id = oi.order_id
  GROUP BY DATE_TRUNC('month', o.order_date)
),
product_agg AS (
  SELECT
    DATE_TRUNC('month', o.order_date) AS month,
    p.product_id,
    p.product_name,
    SUM(oi.quantity) AS total_qty,
    COALESCE(SUM(oi.quantity * oi.unit_price), 0) AS revenue
  FROM orders o
  LEFT JOIN order_items oi ON o.order_id = oi.order_id
  LEFT JOIN products p ON oi.product_id = p.product_id
  GROUP BY DATE_TRUNC('month', o.order_date), p.product_id, p.product_name
),
product_ranked AS (
  SELECT
    pa.*,
    ROW_NUMBER() OVER (PARTITION BY pa.month ORDER BY pa.total_qty DESC, pa.revenue DESC) AS rn
  FROM product_agg pa
),
employee_agg AS (
  SELECT
    DATE_TRUNC('month', o.order_date) AS month,
    e.employee_id,
    e.employee_name,
    COALESCE(SUM(oi.quantity * oi.unit_price), 0) AS revenue
  FROM orders o
  LEFT JOIN order_items oi ON o.order_id = oi.order_id
  LEFT JOIN employees e ON o.employee_id = e.employee_id
  GROUP BY DATE_TRUNC('month', o.order_date), e.employee_id, e.employee_name
),
employee_ranked AS (
  SELECT
    ea.*,
    ROW_NUMBER() OVER (PARTITION BY ea.month ORDER BY ea.revenue DESC) AS rn
  FROM employee_agg ea
),
customer_agg AS (
  SELECT
    DATE_TRUNC('month', o.order_date) AS month,
    c.customer_id,
    c.customer_name,
    COUNT(o.order_id) AS orders_count
  FROM orders o
  LEFT JOIN customers c ON o.customer_id = c.customer_id
  GROUP BY DATE_TRUNC('month', o.order_date), c.customer_id, c.customer_name
),
customer_ranked AS (
  SELECT
    ca.*,
    ROW_NUMBER() OVER (PARTITION BY ca.month ORDER BY ca.orders_count DESC) AS rn
  FROM customer_agg ca
)
SELECT
  mr.month,
  mr.total_revenue,
  pr.product_name AS top_selling_product,
  er.employee_name AS best_performing_employee,
  cr.customer_name AS most_active_customer
FROM month_revenue mr
LEFT JOIN product_ranked pr ON mr.month = pr.month AND pr.rn = 1
LEFT JOIN employee_ranked er ON mr.month = er.month AND er.rn = 1
LEFT JOIN customer_ranked cr ON mr.month = cr.month AND cr.rn = 1
ORDER BY mr.month;
