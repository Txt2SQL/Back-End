SELECT
  c.category_name,
  COUNT(DISTINCT p.product_id) AS total_products,
  COALESCE(SUM(oi.quantity), 0) AS total_sales_quantity,
  (
    SELECT s2.supplier_name
    FROM suppliers s2
    JOIN products p2 ON s2.supplier_id = p2.supplier_id
    LEFT JOIN order_items oi2 ON p2.product_id = oi2.product_id
    WHERE p2.category_id = c.category_id
    GROUP BY s2.supplier_id, s2.supplier_name
    ORDER BY COALESCE(SUM(oi2.quantity), 0) DESC
    LIMIT 1
  ) AS main_supplier
FROM categories c
LEFT JOIN products p ON c.category_id = p.category_id
LEFT JOIN order_items oi ON p.product_id = oi.product_id
