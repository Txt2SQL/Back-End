    p.category_id,
    SUM(oi.quantity * oi.unit_price) AS product_revenue,
    category_revenue.total_category_revenue AS category_revenue,
    (SUM(oi.quantity * oi.unit_price) / category_revenue.total_category_revenue) * 100 AS percentage_contribution
FROM products p
JOIN order_items oi ON p.product_id = oi.product_id
JOIN (
    SELECT 
        p2.category_id,

    FROM products p2
    JOIN order_items oi2 ON p2.product_id = oi2.product_id
    GROUP BY p2.category_id
) category_revenue ON p.category_id = category_revenue.category_id
GROUP BY p.product_name, p.category_id, category_revenue.total_category_revenue
HAVING (SUM(oi.quantity * oi.unit_price) / category_revenue.total_category_revenue) > 0.3;

WITH product_revenue AS (
  SELECT
    p.product_id,
    p.product_name,
    p.category_id,
    COALESCE(SUM(oi.quantity * oi.unit_price), 0) AS product_revenue
  FROM products p
  LEFT JOIN order_items oi ON oi.product_id = p.product_id
  GROUP BY p.product_id, p.product_name, p.category_id
),
category_revenue AS (
  SELECT
    pr.category_id,
    COALESCE(SUM(pr.product_revenue), 0) AS category_revenue
  FROM product_revenue pr
  GROUP BY pr.category_id
)
SELECT
  pr.product_name,
  c.category_name,
  pr.product_revenue,
  cr.category_revenue,
  CASE 
    WHEN cr.category_revenue > 0 THEN (pr.product_revenue / cr.category_revenue) * 100
    ELSE 0
  END AS percentage_contribution
FROM product_revenue pr
JOIN category_revenue cr ON cr.category_id = pr.category_id
JOIN categories c ON c.category_id = pr.category_id
WHERE pr.product_revenue > 0.3 * cr.category_revenue
ORDER BY c.category_name, pr.product_name;
