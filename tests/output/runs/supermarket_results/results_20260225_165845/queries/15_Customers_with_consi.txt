FROM customers c

        SUM(o.total_amount) AS monthly_spending
    FROM orders o
    WHERE YEAR(o.order_date) = 2024

WITH months(m) AS (
  SELECT 1 UNION ALL SELECT 2 UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL
  SELECT 5 UNION ALL SELECT 6 UNION ALL SELECT 7 UNION ALL SELECT 8 UNION ALL
  SELECT 9 UNION ALL SELECT 10 UNION ALL SELECT 11 UNION ALL SELECT 12
),
customer_month_spend AS (
  SELECT
    c.customer_id,
    c.customer_name,
    m.m AS month,
    COALESCE(SUM(oi.quantity * oi.unit_price), 0) AS month_spend
  FROM customers c
  CROSS JOIN months m
  LEFT JOIN orders o
    ON o.customer_id = c.customer_id
    AND EXTRACT(YEAR FROM o.order_date) = 2024
    AND EXTRACT(MONTH FROM o.order_date) = m.m
  LEFT JOIN order_items oi
    ON oi.order_id = o.order_id
  GROUP BY c.customer_id, c.customer_name, m.m
),
customer_avg AS (
  SELECT
    cms.customer_id,
    cms.customer_name,
    AVG(cms.month_spend) AS avg_month_spend
  FROM customer_month_spend cms
  GROUP BY cms.customer_id, cms.customer_name
),
violations AS (
  SELECT
    ca.customer_id,
    SUM(CASE WHEN cms.month_spend < 0.8 * ca.avg_month_spend OR cms.month_spend > 1.2 * ca.avg_month_spend THEN 1 ELSE 0 END) AS violations_count
  FROM customer_month_spend cms
  JOIN customer_avg ca ON ca.customer_id = cms.customer_id
  GROUP BY ca.customer_id
)
SELECT
  ca.customer_name,
  ca.avg_month_spend,
  CASE WHEN v.violations_count = 0 THEN 'Consistent' ELSE 'Inconsistent' END AS consistency_flag
FROM customer_avg ca
JOIN violations v ON v.customer_id = ca.customer_id
WHERE v.violations_count = 0
ORDER BY ca.customer_name;
