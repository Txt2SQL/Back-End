Query:
WITH prod_orders AS (
  SELECT
    oi.product_id,
    p.product_name,
    COUNT(DISTINCT oi.order_id) AS orders_count
  FROM order_items oi
  JOIN products p ON p.product_id = oi.product_id
  GROUP BY oi.product_id, p.product_name
),
pairs AS (
  SELECT
    oi1.product_id AS product_id_a,
    oi2.product_id AS product_id_b,
    COUNT(DISTINCT oi1.order_id) AS together_count
  FROM order_items oi1
  JOIN order_items oi2
    ON oi1.order_id = oi2.order_id
    AND oi1.product_id < oi2.product_id
  GROUP BY oi1.product_id, oi2.product_id
)
SELECT
  pa.product_name AS product_a,
  pb.product_name AS product_b,
  p.together_count AS times_together
FROM pairs p
JOIN prod_orders pa ON pa.product_id = p.product_id_a
JOIN prod_orders pb ON pb.product_id = p.product_id_b
WHERE p.together_count = pa.orders_count
  AND p.together_count = pb.orders_count
ORDER BY pa.product_name, pb.product_name;
