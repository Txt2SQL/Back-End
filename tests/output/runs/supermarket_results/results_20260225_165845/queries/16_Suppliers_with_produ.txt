    s.supplier_name,

    SUM(CASE WHEN p.price > 50 THEN 1 ELSE 0 END) AS premium_count
FROM suppliers s
JOIN products p ON s.supplier_id = p.supplier_id
GROUP BY s.supplier_name
HAVING
