Customers who spent more than average. Find customers whose total spending is above the average customer spending

SELECT c.customer_id, c.customer_name, SUM(p.amount) AS total_spending
FROM customers c
JOIN purchases p ON c.customer_id = p.customer_id
GROUP BY c.customer_id, c.customer_name
HAVING SUM(p.amount) > (SELECT AVG(total_spending) 
                        FROM (SELECT SUM(p1.amount) AS total_spending
                              FROM purchases p1
                              GROUP BY p1.customer_id) AS average_spending);
