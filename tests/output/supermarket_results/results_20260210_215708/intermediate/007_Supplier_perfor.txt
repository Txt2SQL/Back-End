1. Request: Supplier performance analysis. For each supplier, show: Total products supplied, total quantity sold, and total revenue generated, Include suppliers with no sales

--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

🤖 Model: gpt-4o

🧮 Query:


SELECT 
    s.supplier_id,
    s.supplier_name,
    COUNT(p.product_id) AS total_products_supplied,
    COALESCE(SUM(i.quantity_change), 0) AS total_quantity_sold,
    COALESCE(SUM(i.quantity_change * p.price), 0) AS total_revenue_generated
FROM 
    suppliers s
LEFT JOIN 
    products p ON s.supplier_id = p.supplier_id
LEFT JOIN 
    inventory i ON p.product_id = i.product_id AND i.change_type = 'sale'
GROUP BY 
    s.supplier_id, s.supplier_name;

  suppliers.supplier_name,
  COUNT(DISTINCT products.product_id) AS total_products_supplied,
  COALESCE(SUM(CASE WHEN inventory.change_type = 'sale' THEN inventory.quantity_change ELSE 0 END), 0) AS total_quantity_sold,
  COALESCE(SUM(CASE WHEN inventory.change_type = 'sale' THEN inventory.quantity_change * products.price ELSE 0 END), 0) AS total_revenue_generated
FROM suppliers
LEFT JOIN products ON products.supplier_id = suppliers.supplier_id
LEFT JOIN inventory ON inventory.product_id = products.product_id
GROUP BY suppliers.supplier_id, suppliers.supplier_name
ORDER BY suppliers.supplier_name;
