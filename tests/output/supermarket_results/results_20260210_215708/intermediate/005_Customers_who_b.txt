1. Request: Customers who bought the most popular product. Find customers who purchased the product that has been sold the most times

--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

🤖 Model: gpt-4o

🧮 Query:


SELECT CONCAT(customers.first_name, ' ', customers.last_name) AS customer_name, customers.email
FROM reviews
JOIN customers ON reviews.customer_id = customers.customer_id
WHERE reviews.product_id = (
    SELECT product_id
    FROM reviews
    GROUP BY product_id
    ORDER BY COUNT(product_id) DESC
    LIMIT 1
)
GROUP BY customers.customer_id, customers.first_name, customers.last_name, customers.email;

🤖 Model: gpt-5-mini

🧮 Query:


SELECT DISTINCT
    c.customer_id,
    c.first_name,
    c.last_name,
    c.email,
    p.product_id,
    p.product_name
FROM reviews r
JOIN customers c ON r.customer_id = c.customer_id
JOIN products p ON r.product_id = p.product_id
JOIN (
    SELECT product_id
    FROM reviews
    GROUP BY product_id
    HAVING COUNT(*) = (
        SELECT MAX(product_count) FROM (
            SELECT COUNT(*) AS product_count FROM reviews GROUP BY product_id
        ) AS counts_sub
    )
) top_p ON r.product_id = top_p.product_id;
